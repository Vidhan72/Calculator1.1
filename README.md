num_1 = int(input("Enter First number: "))
Op = input("Enter Operator(+,-,*,/): ")
num_2 = int(input("Enter sec number: "))

if Op == "+":
    print(num_1 + num_2)

elif Op == "-":
    print(num_1 - num_2)

elif Op == "*":
    print(num_1 * num_2)

elif Op == "/":
    print(num_1 / num_2)

else:
    print ("**Invalid Operator**")
